圆角

-webkit-border-radius: 10px;
   -moz-border-radius: 10px;
        border-radius: 10px;

/* 标准写法 */
border-top-left-radius: 10px;
border-top-right-radius: 20px; 
border-bottom-right-radius: 30px; 
border-bottom-left-radius: 0;


/* Chrome and Safari 老版本支持 */ 
-webkit-border-top-left-radius: 10px;
-webkit-border-top-right-radius: 20px; 
-webkit-border-bottom-right-radius: 30px; 
-webkit-border-bottom-left-radius: 0;

/* Firefox 老版本支持 */
-moz-border-radius-topleft: 10px;
-moz-border-radius-topright: 20px;
-moz-border-radius-bottomright: 30px;
-moz-border-radius-bottomleft: 0;

渐变

filter: progid:DXImageTransform.Microsoft.gradient(startColorstr='#c88fdf', endColorstr='#9b59b6'); /* IE */
background: -webkit-gradient(linear, left top, left bottom, from(#c88fdf), to(#9b59b6));
background: -moz-linear-gradient(top, #c88fdf, #9b59b6);
background: linear-gradient(top, #c88fdf, #9b59b6);

旋转

-webkit-transform: rotate(7.5deg);
   -moz-transform: rotate(7.5deg);
    -ms-transform: rotate(7.5deg);
        transform: rotate(7.5deg);

/* 下面其中的x只能取值为1，2，3，0，分别表示顺时针选择90度、180度、270度和360度。 */
filter: progid:DXImageTransform.Microsoft.BasicImage(rotation=x);
-ms-filter: "progid:DXImageTransform.Microsoft.BasicImage(rotation=x)";

Web Fonts!
@font-face {
  font-family: 'MyFont';
  src: url('myfont.eot'); /* IE6+ */
  src: local('myfont.ttf'),
  url('myfont.woff') format('woff'), /* FF3.6 */
  url('myfont.ttf') format('truetype'); /* FF3.5+, Saf3+,Chrome,Opera10+ */
}
